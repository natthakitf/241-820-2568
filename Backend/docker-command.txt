// Show all containers (default shows just running)
docker ps --all

// Show just running containers
docker ps

docker run hello-world

// Stop containers
docker stop <containers_id>

// pull an image from a registry
docker pull <image_name>

// create a containers from an image
docker create <image_name>

// start a containers
docker start <containers_id>

// remove a containers
docker rm <containers_id>

// remove all containers
docker system prune

// nginx server
docker run nginx

// sh containers
docker run it <image_name or containers_id> sh

// cat file inside containers
cat filename.txt


docker run -p 80:80 nginx
docker run -p 9000 nginx

//foreground mode
docker run -p 9000:80 nginx

//detached mode
docker run -d -p 9100:80 nginx
